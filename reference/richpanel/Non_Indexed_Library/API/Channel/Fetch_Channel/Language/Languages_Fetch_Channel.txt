[Language: Node]
import richpanel from '@api/richpanel';

richpanel.auth('test');
richpanel.fetchAChannel({id: 'id'})
  .then(({ data }) => console.log(data))
  .catch(err => console.error(err));


[Language: Ruby]
require 'uri'
require 'net/http'

url = URI("https://api.richpanel.com/v1/channel/id")

http = Net::HTTP.new(url.host, url.port)
http.use_ssl = true

request = Net::HTTP::Get.new(url)
request["accept"] = 'application/json'
request["x-richpanel-key"] = 'test'

response = http.request(request)
puts response.read_body
