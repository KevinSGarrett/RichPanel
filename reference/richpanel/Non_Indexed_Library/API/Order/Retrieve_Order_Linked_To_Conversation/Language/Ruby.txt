Retrieve an Order
get
https://api.richpanel.com/v1/order/{appclient_id}/{order_id}

Language: Ruby

require 'uri'
require 'net/http'

url = URI("https://api.richpanel.com/v1/order/conversationId")

http = Net::HTTP.new(url.host, url.port)
http.use_ssl = true

request = Net::HTTP::Get.new(url)
request["accept"] = 'application/json'
request["x-richpanel-key"] = 'test'

response = http.request(request)
puts response.read_body
