Language: Python

import requests

url = "https://api.richpanel.com/v1/order/appclient_id"

headers = {
    "accept": "application/json",
    "content-type": "application/json",
    "x-richpanel-key": "test"
}

response = requests.post(url, headers=headers)

print(response.text)
