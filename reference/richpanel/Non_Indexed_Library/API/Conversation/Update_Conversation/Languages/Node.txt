Update a Conversation

Language: Node

import richpanel from '@api/richpanel';

richpanel.auth('test');
richpanel.updateAConversation({
  ticket: {
    status: 'OPEN',
    comment: {public: true},
    via: {channel: 'email'},
    assignee_id: 'unassigned'
  }
}, {id: 'id'})
  .then(({ data }) => console.log(data))
  .catch(err => console.error(err));
