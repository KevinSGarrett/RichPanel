Retrieve Conversations using Customer's Email or Phone
get
https://api.richpanel.com/v1/tickets/{type}/{id}

Language: PHP

<?php
require_once('vendor/autoload.php');

$client = new \GuzzleHttp\Client();

$response = $client->request('GET', 'https://api.richpanel.com/v1/tickets/phone/id', [
  'headers' => [
    'accept' => 'application/json',
    'x-richpanel-key' => 'test',
  ],
]);

echo $response->getBody();
