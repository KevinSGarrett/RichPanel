Retrieve Conversations using Customer's Email or Phone
get
https://api.richpanel.com/v1/tickets/{type}/{id}

Language: Python

import requests

url = "https://api.richpanel.com/v1/tickets/phone/id"

headers = {
    "accept": "application/json",
    "x-richpanel-key": "test"
}

response = requests.get(url, headers=headers)

print(response.text)
