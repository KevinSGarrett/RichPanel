Retrieve Conversations using Customer's Email or Phone
get
https://api.richpanel.com/v1/tickets/{type}/{id}

Language: RUBY

require 'uri'
require 'net/http'

url = URI("https://api.richpanel.com/v1/tickets/phone/id")

http = Net::HTTP.new(url.host, url.port)
http.use_ssl = true

request = Net::HTTP::Get.new(url)
request["accept"] = 'application/json'
request["x-richpanel-key"] = 'test'

response = http.request(request)
puts response.read_body
