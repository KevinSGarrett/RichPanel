import requests

url = "https://api.richpanel.com/v1/tickets/id/remove-tags"

headers = {
    "accept": "application/json",
    "content-type": "application/json",
    "x-richpanel-key": "test"
}

response = requests.put(url, headers=headers)

print(response.text)
