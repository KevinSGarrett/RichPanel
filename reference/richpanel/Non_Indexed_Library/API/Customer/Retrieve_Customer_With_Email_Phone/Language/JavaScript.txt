Retrieve a Customer with Email/Phone
get
https://api.richpanel.com/v1/customers/{type}/{id}

Language: JavaScript

const options = {
  method: 'GET',
  headers: {accept: 'application/json', 'x-richpanel-key': 'test'}
};

fetch('https://api.richpanel.com/v1/customers/type/id', options)
  .then(res => res.json())
  .then(res => console.log(res))
  .catch(err => console.error(err));
